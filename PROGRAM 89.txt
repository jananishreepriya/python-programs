# Python program to generate magic square without numpy

order = int(input("Enter order of magic square (order must be odd): "))

# If even number is given, it will be incremented by one
if order % 2 == 0:
    order = order + 1
    print("Given order is even so it is incremented by 1.")

# Initialize the magic square
magic = [[0] * order for _ in range(order)]

mid = order // 2
k = mid
j = 0

for i in range(1, order * order + 1):
    magic[j][k] = i
    p = j
    j = j - 1
    q = k
    k = k + 1

    if j < 0:
        j = order - 1

    if k > order - 1:
        k = 0

    if magic[j][k] != 0:
        k = q
        j = p + 1

print("Generated Magic Square is: \n")

# Printing the magic square
for j in range(order):
    print("|", end="")
    for k in range(order):
        print("%4d |" % magic[j][k], end="")
    print()

# Printing the bottom line
for i in range(1, 6 * order + 1):
    print("-", end="")

print()
